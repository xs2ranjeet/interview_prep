#include <bits/stdc++.h>
using namespace std;

int romanToInt(string s){
    unordered_map<char,int> val{{'I',1},{'V',5},{'X',10},{'L',50},{'C',100},{'D',500},{'M',1000}};
    int n = s.length();
    int res = 0;
    for(int i = 0; i < n; i++) {
        int v = val[s[i]];
        if(i+1 < n and val[s[i+1]] > v)
            res -= v;
        else
            res += v;
    }
    return res;
}
